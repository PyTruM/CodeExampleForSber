SELECT * 
FROM bank.данные;

SELECT count(*) 
FROM bank.данные 
where time(`Время регистрации`) >= '09:00' 
and time(`Время регистрации`) <= '18:00';

select count(*) as `Количество заявок` 
from bank.данные 
where `Рабочая группа` = 'Группа обработки заявок' 
and `Система` in ('РЗД', 'СУД')
and time(`Время регистрации`) >= '09:00' 
and time(`Время регистрации`) <= '18:00';

select round((select count(*) 
from bank.данные 
where `Сервисный запрос` in ('Неопределенный запрос', 'Неопределенный инцидент') 
and `Рабочая группа` = 'Группа обработки заявок' 
and `Источник` in ('Телефон', 'Почта')) * 100.0 / 
(select count(*) 
from bank.данные 
where `Рабочая группа` = 'Группа обработки заявок' 
and `Источник` in ('Телефон', 'Почта')), 2)  as `Доля`
from bank.данные limit 1;

select count(*) 
from bank.данные 
where `Сервисный запрос` in ('Неопределенный запрос', 'Неопределенный инцидент') 
and `Рабочая группа` = 'Группа обработки заявок' and `Источник` in ('Телефон', 'Почта');

select count(*) 
from bank.данные 
where `Рабочая группа` = 'Группа обработки заявок' 
and `Источник` in ('Телефон', 'Почта');

select round((select count(*) from bank.данные 
where `Стандартное решение` != '' and `Статус` in ('Завершено','Закрыто') 
and `Обращение маршрутизировано` = 'false' 
and `Рабочая группа` like 'Группа обработки%' 
and `Источник` in ('Телефон', 'Почта')) * 100.0 /
(select count(*) 
from bank.данные 
where `Обращение маршрутизировано` = 'false' 
and `Рабочая группа` like 'Группа обработки%' 
and `Источник` in ('Телефон', 'Почта')), 2) as `Доля` 
from bank.данные limit 1;

select count(*) 
from bank.данные 
where `Стандартное решение` != '' 
and `Статус` in ('Завершено','Закрыто') 
and `Обращение маршрутизировано` = 'false' 
and `Рабочая группа` like 'Группа обработки%' 
and `Источник` in ('Телефон', 'Почта');

select count(*) 
from bank.данные 
where `Обращение маршрутизировано` = 'false' 
and `Рабочая группа` like 'Группа обработки%' 
and `Источник` in ('Телефон', 'Почта');

select * 
from bank.данные 
where `Рабочая группа` 
like 'Группа%';
